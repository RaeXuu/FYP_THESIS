**Fig 3.2 — 预处理各步骤波形图**（代码如下，取一条 Normal 录音跑完整 pipeline）

```python
import numpy as np
import matplotlib.pyplot as plt
import librosa
from scipy.signal import butter, filtfilt

# ---- 参数（与论文一致） ----
SR = 2000
LOWCUT, HIGHCUT = 25, 400
SEG_LEN = SR * 2          # 2s window
N_MELS, N_FFT, HOP = 64, 256, 128

# ---- 载入一条 Normal 录音（替换路径）----
raw, orig_sr = librosa.load('path/to/normal_sample.wav', sr=SR, mono=True)

# Step 1: raw waveform（截前 4s 方便展示）
display_len = SR * 4
raw_display = raw[:display_len]
t = np.arange(len(raw_display)) / SR

# Step 2: 带通滤波
b, a = butter(4, [LOWCUT / (SR/2), HIGHCUT / (SR/2)], btype='band')
filtered = filtfilt(b, a, raw)
filtered_display = filtered[:display_len]

# Step 3: 取第一个 2s 分段
segment = filtered[:SEG_LEN]
t_seg = np.arange(SEG_LEN) / SR

# Step 4: Log-Mel 频谱
mel = librosa.feature.melspectrogram(y=segment, sr=SR,
                                      n_fft=N_FFT, hop_length=HOP,
                                      n_mels=N_MELS, fmin=LOWCUT, fmax=HIGHCUT)
log_mel = librosa.power_to_db(mel, ref=np.max)

# ---- 画四格图 ----
fig, axes = plt.subplots(4, 1, figsize=(10, 10))
plt.rcParams['font.family'] = 'serif'

axes[0].plot(t, raw_display, linewidth=0.6)
axes[0].set_title('(a) Raw waveform')
axes[0].set_ylabel('Amplitude'); axes[0].set_xlabel('Time (s)')

axes[1].plot(t, filtered_display, linewidth=0.6, color='steelblue')
axes[1].set_title('(b) After bandpass filter (25–400 Hz)')
axes[1].set_ylabel('Amplitude'); axes[1].set_xlabel('Time (s)')

axes[2].plot(t_seg, segment, linewidth=0.6, color='darkorange')
axes[2].set_title('(c) Segmented window (2 s)')
axes[2].set_ylabel('Amplitude'); axes[2].set_xlabel('Time (s)')

img = axes[3].imshow(log_mel, aspect='auto', origin='lower',
                     extent=[0, segment.shape[0]/SR, 0, N_MELS],
                     cmap='magma')
axes[3].set_title('(d) Log-Mel spectrogram (32×64)')
axes[3].set_ylabel('Mel bin'); axes[3].set_xlabel('Time (s)')
fig.colorbar(img, ax=axes[3], label='dB')

plt.tight_layout()
plt.savefig('fig3_2_preprocessing_steps.png', dpi=300, bbox_inches='tight')
```

**Fig 3.3 — Normal vs Abnormal Log-Mel 频谱对比**：需从测试集各取一条代表性录音，走完预处理 pipeline 后画热图。

```python
import numpy as np
import matplotlib.pyplot as plt
import librosa
from scipy.signal import butter, filtfilt

SR = 2000
LOWCUT, HIGHCUT = 25, 400
SEG_LEN = SR * 2
N_MELS, N_FFT, HOP = 64, 256, 128

def get_log_mel(path):
    raw, _ = librosa.load(path, sr=SR, mono=True)
    b, a = butter(4, [LOWCUT / (SR/2), HIGHCUT / (SR/2)], btype='band')
    filtered = filtfilt(b, a, raw)
    seg = filtered[:SEG_LEN]
    mel = librosa.feature.melspectrogram(y=seg, sr=SR,
                                          n_fft=N_FFT, hop_length=HOP,
                                          n_mels=N_MELS, fmin=LOWCUT, fmax=HIGHCUT)
    return librosa.power_to_db(mel, ref=np.max)

normal_mel   = get_log_mel('path/to/normal_sample.wav')    # 替换路径
abnormal_mel = get_log_mel('path/to/abnormal_sample.wav')  # 替换路径

fig, axes = plt.subplots(1, 2, figsize=(10, 4))
plt.rcParams['font.family'] = 'serif'

for ax, mel, title in zip(axes,
                           [normal_mel, abnormal_mel],
                           ['(a) Normal', '(b) Abnormal']):
    img = ax.imshow(mel, aspect='auto', origin='lower',
                    extent=[0, SEG_LEN/SR, 0, N_MELS],
                    cmap='magma', vmin=-80, vmax=0)
    ax.set_title(title)
    ax.set_xlabel('Time (s)')
    ax.set_ylabel('Mel bin')
    fig.colorbar(img, ax=ax, label='dB')

plt.tight_layout()
plt.savefig('fig3_3_mel_comparison.png', dpi=300, bbox_inches='tight')
```

自动展示 Coordinate Attention 内部结构和残差连接（分叉后汇合的箭头）。

```python

model = YourCNNModel()

y = model(x)

dot = make_dot(y, params=dict(model.named_parameters()))
```

## draw.io 画法建议

针对 CNN + Coordinate Attention + 残差连接：

- 用矩形块表示卷积层、BN、激活函数
- Coordinate Attention 模块单独用不同颜色的子图，内部结构：
  - H/W 方向 Average Pool → Concat → FC → Split → Sigmoid
- 残差连接用**跳跃箭头**绕过主路径后相加
